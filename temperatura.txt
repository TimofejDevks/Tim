temperaturos = [15.5, 17.0, 16.3, 18.2, 14.5, 19.1, 20.0]

bendras = 0
dienu_skaicius = 0

for temp in temperaturos:
    bendras += temp  
    dienu_skaicius += 1  

vidutine = bendras / dienu_skaicius

print("Vidutinė savaitės temperatūra:", vidutine)

print("Vidutinė savaitės temperatūra:", round(vidutine, 2))
